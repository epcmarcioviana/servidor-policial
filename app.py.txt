import os
import json
from flask import Flask, request, jsonify
import google.generativeai as genai

app = Flask(__name__)

# Puxa a chave de forma segura
chave_api = os.environ.get("GEMINI_API_KEY", "").strip()
genai.configure(api_key=chave_api)

@app.route('/')
def home():
    return "Servidor Policial Ativo!", 200

@app.route('/processar-pdf', methods=['POST'])
def processar_pdf():
    try:
        dados_recebidos = request.get_json()
        if not dados_recebidos or 'base64_data' not in dados_recebidos:
            return jsonify({"erro": "Dados binários ausentes"}), 400
        
        base64_pdf = dados_recebidos['base64_data']
        
        documento = {
            "mime_type": "application/pdf",
            "data": base64_pdf
        }
        
        prompt = (
            "Extraia as informações deste Termo de Apreensão Policial e retorne APENAS "
            "um objeto JSON válido. Campos obrigatórios: bo_sinesp, data_apreensao, "
            "tipo_objeto, descricao_detalhada, quantidade, envolvido."
        )
        
        model = genai.GenerativeModel("gemini-1.5-flash")
        response = model.generate_content([prompt, documento])
        
        texto_limpo = response.text.strip()
        if texto_limpo.startswith("```"):
            texto_limpo = texto_limpo.replace("```json", "").replace("```", "").strip()
            
        return jsonify(json.loads(texto_limpo)), 200
        
    except Exception as e:
        return jsonify({"erro": str(e)}), 500

if __name__ == '__main__':
    # Garante compatibilidade local caso necessário
    app.run(host='0.0.0.0', port=5000)
